def initialize(
  text: nil,
  anchor: nil,
  level: nil,
  link: nil
)
  raise ArgumentError if !text.kind_of?(String)
  @text = text
  @anchor = anchor || self.class.text_to_anchor(text)
  @level = level
  @link = link
end

def self.parse_line_level(line)
  return line =~/^\s*([#=]+)/ ? $1.length : nil
end

def self.trim_line(line)
